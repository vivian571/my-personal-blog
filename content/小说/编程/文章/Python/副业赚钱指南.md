比如，这个简单的Excel数据处理脚本，就能帮助财务人员节省几小时的工作时间：

```python
# excel_automation.py
import pandas as pd
import numpy as np
from datetime import datetime

def process_financial_data(input_file, output_file):
    """自动处理财务数据并生成报表"""
    # 读取Excel文件
    df = pd.read_excel(input_file)
    
    # 数据清洗和处理
    df['日期'] = pd.to_datetime(df['日期'])
    df['月份'] = df['日期'].dt.month
    df['季度'] = df['日期'].dt.quarter
    
    # 计算关键指标
    monthly_summary = df.groupby('月份')['金额'].agg(['sum', 'mean', 'min', 'max'])
    quarterly_summary = df.groupby('季度')['金额'].agg(['sum', 'mean', 'min', 'max'])
    
    # 创建一个Excel写入器
    with pd.ExcelWriter(output_file) as writer:
        # 写入原始数据
        df.to_excel(writer, sheet_name='原始数据', index=False)
        
        # 写入月度汇总
        monthly_summary.to_excel(writer, sheet_name='月度汇总')
        
        # 写入季度汇总
        quarterly_summary.to_excel(writer, sheet_name='季度汇总')
        
    print(f"报表已生成: {output_file}")

if __name__ == "__main__":
    process_financial_data("财务数据.xlsx", "财务报表_" + datetime.now().strftime("%Y%m%d") + ".xlsx")
```

```python
# web_traffic_analyzer.py
import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns

class WebTrafficAnalyzer:
    def __init__(self, data_file):
        self.data = pd.read_csv(data_file)
        self.prepare_data()
        
    def prepare_data(self):
        # 转换时间列为datetime类型
        self.data['访问时间'] = pd.to_datetime(self.data['访问时间'])
        self.data['小时'] = self.data['访问时间'].dt.hour
        self.data['星期'] = self.data['访问时间'].dt.day_name()
        self.data['日期'] = self.data['访问时间'].dt.date
    
    def generate_daily_report(self, output_file="流量日报.html"):
        # 每日访问量统计
        daily_visits = self.data.groupby('日期').size().reset_index(name='访问量')
        
        # 转化率计算
        conversions = self.data[self.data['是否转化'] == True].groupby('日期').size().reset_index(name='转化数')
        daily_report = pd.merge(daily_visits, conversions, on='日期', how='left')
        daily_report['转化率'] = (daily_report['转化数'] / daily_report['访问量'] * 100).round(2)
        
        # 生成HTML报告
        html_content = f"""
        <html>
        <head>
            <title>网站流量日报</title>
            <style>
                body {{ font-family: Arial, sans-serif; margin: 20px; }}
                table {{ border-collapse: collapse; width: 100%; }}
                th, td {{ border: 1px solid #ddd; padding: 8px; text-align: left; }}
                th {{ background-color: #f2f2f2; }}
                tr:nth-child(even) {{ background-color: #f9f9f9; }}
            </style>
        </head>
        <body>
            <h1>网站流量日报</h1>
            {daily_report.to_html(index=False)}
        </body>
        </html>
        """
        
        with open(output_file, 'w', encoding='utf-8') as f:
            f.write(html_content)
        
        return daily_report
    
    def plot_hourly_distribution(self, output_file="每小时流量分布.png"):
        plt.figure(figsize=(12, 6))
        sns.countplot(x='小时', data=self.data, palette='viridis')
        plt.title('网站每小时访问量分布')
        plt.xlabel('小时')
        plt.ylabel('访问量')
        plt.savefig(output_file)
        plt.close()

# 使用示例
if __name__ == "__main__":
    analyzer = WebTrafficAnalyzer("网站访问日志.csv")
    daily_report = analyzer.generate_daily_report()
    analyzer.plot_hourly_distribution()
    print("分析报告已生成完毕！")
```

```python
# wechat_miniapp_backend.py
from flask import Flask, request, jsonify
import pymysql
import hashlib
import time
import jwt

app = Flask(__name__)
app.config['SECRET_KEY'] = 'your_secret_key'

# 数据库连接
def get_db_connection():
    return pymysql.connect(
        host='localhost',
        user='root',
        password='password',
        database='miniapp_db',
        charset='utf8mb4',
        cursorclass=pymysql.cursors.DictCursor
    )

# 用户登录
@app.route('/api/login', methods=['POST'])
def login():
    data = request.get_json()
    username = data.get('username')
    password = data.get('password')
    
    if not username or not password:
        return jsonify({'error': '用户名和密码不能为空'}), 400
    
    # 密码加密
    hashed_password = hashlib.md5(password.encode()).hexdigest()
    
    conn = get_db_connection()
    cursor = conn.cursor()
    
    try:
        cursor.execute('SELECT * FROM users WHERE username = %s AND password = %s', (username, hashed_password))
        user = cursor.fetchone()
        
        if user:
            # 生成JWT令牌
            token = jwt.encode({
                'user_id': user['id'],
                'username': user['username'],
                'exp': int(time.time()) + 86400  # 24小时过期
            }, app.config['SECRET_KEY'], algorithm='HS256')
            
            return jsonify({
                'token': token,
                'user_id': user['id'],
                'username': user['username']
            })
        else:
            return jsonify({'error': '用户名或密码错误'}), 401
    finally:
        cursor.close()
        conn.close()

# 获取商品列表
@app.route('/api/products', methods=['GET'])
def get_products():
    conn = get_db_connection()
    cursor = conn.cursor()
    
    try:
        cursor.execute('SELECT * FROM products WHERE status = 1 ORDER BY created_at DESC')
        products = cursor.fetchall()
        return jsonify({'products': products})
    finally:
        cursor.close()
        conn.close()

if __name__ == '__main__':
    app.run(debug=True, host='0.0.0.0', port=5000)
```

```python
# course_platform.py
from flask import Flask, render_template, request, redirect, url_for, session
import sqlite3
import os
from werkzeug.utils import secure_filename

app = Flask(__name__)
app.secret_key = 'your_secret_key'
app.config['UPLOAD_FOLDER'] = 'static/uploads'

# 确保上传目录存在
os.makedirs(app.config['UPLOAD_FOLDER'], exist_ok=True)

# 初始化数据库
def init_db():
    conn = sqlite3.connect('courses.db')
    cursor = conn.cursor()
    cursor.execute('''
    CREATE TABLE IF NOT EXISTS courses (
        id INTEGER PRIMARY KEY AUTOINCREMENT,
        title TEXT NOT NULL,
        description TEXT,
        price REAL NOT NULL,
        image_path TEXT,
        content TEXT
    )
    ''')
    conn.commit()
    conn.close()

@app.route('/')
def home():
    conn = sqlite3.connect('courses.db')
    cursor = conn.cursor()
    cursor.execute('SELECT id, title, description, price, image_path FROM courses')
    courses = cursor.fetchall()
    conn.close()
    return render_template('home.html', courses=courses)

@app.route('/course/<int:course_id>')
def course_detail(course_id):
    conn = sqlite3.connect('courses.db')
    cursor = conn.cursor()
    cursor.execute('SELECT * FROM courses WHERE id = ?', (course_id,))
    course = cursor.fetchone()
    conn.close()
    
    if course:
        return render_template('course_detail.html', course=course)
    return redirect(url_for('home'))

@app.route('/admin/add_course', methods=['GET', 'POST'])
def add_course():
    if request.method == 'POST':
        title = request.form['title']
        description = request.form['description']
        price = float(request.form['price'])
        content = request.form['content']
        
        # 处理图片上传
        image_path = ''
        if 'image' in request.files:
            file = request.files['image']
            if file.filename != '':
                filename = secure_filename(file.filename)
                file.save(os.path.join(app.config['UPLOAD_FOLDER'], filename))
                image_path = 'uploads/' + filename
        
        conn = sqlite3.connect('courses.db')
        cursor = conn.cursor()
        cursor.execute(
            'INSERT INTO courses (title, description, price, image_path, content) VALUES (?, ?, ?, ?, ?)',
            (title, description, price, image_path, content)
        )
        conn.commit()
        conn.close()
        
        return redirect(url_for('home'))
    
    return render_template('add_course.html')

if __name__ == '__main__':
    init_db()
    app.run(debug=True)
```

以下是一个简单的WordPress SEO优化插件示例：

```php
<?php
/**
 * Plugin Name: 简易SEO优化工具
 * Plugin URI: https://your-website.com/simple-seo-tool

 * Author URI: https://your-website.com
 * License: GPL2
 */

// 防止直接访问
if (!defined('ABSPATH')) {
    exit;
}

class SimpleSeoTool {
    public function __construct() {
        // 添加元数据框到文章编辑页面
        add_action('add_meta_boxes', array($this, 'add_meta_boxes'));
        // 保存元数据
        add_action('save_post', array($this, 'save_meta_data'));
        // 修改标题标签
        add_filter('document_title_parts', array($this, 'custom_document_title'));
        // 添加元描述
        add_action('wp_head', array($this, 'add_meta_description'));
        // 添加管理菜单
        add_action('admin_menu', array($this, 'add_admin_menu'));
    }
    
    // 添加元数据框
    public function add_meta_boxes() {
        add_meta_box(
            'simple_seo_tool',
            'SEO设置',
            array($this, 'meta_box_callback'),
            'post',
            'normal',
            'high'
        );
    }
    
    // 元数据框回调函数
    public function meta_box_callback($post) {
        wp_nonce_field('simple_seo_tool_nonce', 'simple_seo_tool_nonce');
        $seo_title = get_post_meta($post->ID, '_seo_title', true);
        $seo_description = get_post_meta($post->ID, '_seo_description', true);
        $seo_keywords = get_post_meta($post->ID, '_seo_keywords', true);
        
        echo '<p><label for="seo_title">SEO标题:</label></p>';
        echo '<p><input type="text" id="seo_title" name="seo_title" value="' . esc_attr($seo_title) . '" style="width:100%" /></p>';
        
        echo '<p><label for="seo_description">SEO描述:</label></p>';
        echo '<p><textarea id="seo_description" name="seo_description" style="width:100%" rows="3">' . esc_textarea($seo_description) . '</textarea></p>';
        
        echo '<p><label for="seo_keywords">SEO关键词 (用逗号分隔):</label></p>';
        echo '<p><input type="text" id="seo_keywords" name="seo_keywords" value="' . esc_attr($seo_keywords) . '" style="width:100%" /></p>';
    }
    
    // 保存元数据
    public function save_meta_data($post_id) {
        if (!isset($_POST['simple_seo_tool_nonce']) || !wp_verify_nonce($_POST['simple_seo_tool_nonce'], 'simple_seo_tool_nonce')) {
            return;
        }
        
        if (defined('DOING_AUTOSAVE') && DOING_AUTOSAVE) {
            return;
        }
        
        if (!current_user_can('edit_post', $post_id)) {
            return;
        }
        
        if (isset($_POST['seo_title'])) {
            update_post_meta($post_id, '_seo_title', sanitize_text_field($_POST['seo_title']));
        }
        
        if (isset($_POST['seo_description'])) {
            update_post_meta($post_id, '_seo_description', sanitize_textarea_field($_POST['seo_description']));
        }
        
        if (isset($_POST['seo_keywords'])) {
            update_post_meta($post_id, '_seo_keywords', sanitize_text_field($_POST['seo_keywords']));
        }
    }
    
    // 自定义文档标题
    public function custom_document_title($title_parts) {
        if (is_singular('post')) {
            global $post;
            $seo_title = get_post_meta($post->ID, '_seo_title', true);
            if (!empty($seo_title)) {
                $title_parts['title'] = $seo_title;
            }
        }
        return $title_parts;
    }
    
    // 添加元描述
    public function add_meta_description() {
        if (is_singular('post')) {
            global $post;
            $seo_description = get_post_meta($post->ID, '_seo_description', true);
            $seo_keywords = get_post_meta($post->ID, '_seo_keywords', true);
            
            if (!empty($seo_description)) {
                echo '<meta name="description" content="' . esc_attr($seo_description) . '" />' . "\n";
            }
            
            if (!empty($seo_keywords)) {
                echo '<meta name="keywords" content="' . esc_attr($seo_keywords) . '" />' . "\n";
            }
        }
    }
    
    // 添加管理菜单
    public function add_admin_menu() {
        add_options_page(
            'SEO优化工具设置',
            'SEO优化工具',
            'manage_options',
            'simple-seo-tool',
            array($this, 'admin_page_callback')
        );
    }
    
    // 管理页面回调函数
    public function admin_page_callback() {
        echo '<div class="wrap">';
        echo '<h1>SEO优化工具设置</h1>';
        echo '<p>使用此插件可以为您的文章添加SEO标题、描述和关键词，提升网站在搜索引擎中的排名。</p>';
        echo '<p>在编辑文章时，您可以在文章编辑页面右侧找到"SEO设置"面板，填写相关信息。</p>';
        
        // 显示全局设置表单
        echo '<form method="post" action="options.php">';

        settings_fields('simple_seo_tool_options');
        do_settings_sections('simple-seo-tool');
        
        // 提交按钮

        echo '</form>';
        echo '</div>';

    public static function init() {
        $instance = new self();
    }
}
